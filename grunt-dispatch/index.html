grunt-dispatch

  • Grunt tasks for copying the necessary files of package/components directory to your src/build directory

Config Example


dispatch: {
    options: {
        directory: "bower_components"
    },
    "ozjs": {
        use: {
            "js/lib/": "oz.js"
        }
    },
    "mo": {
        use: {
            "js/mod/mo/": ["**/*.js", "!**/Gruntfile.js"]
        }
    },
    "moui": {
        use: [{
            cwd: "css/moui",
            src: ["**"],
            dest: "css/moui/"
        }, {
            cwd: "asset",
            src: ["**"],
            dest: "pics/"
        }, {
            src: ["**/*.js", "!**/Gruntfile.js"],
            dest: "js/mod/moui/"
        }]
    }
},
